      // 正则 实现去除文件后缀名
      function splitFileName(fileName) {
        var pattern = /\.{1}[a-z]{1,}$/;
        if (pattern.exec(fileName) !== null) {
          return (fileName.slice(0, pattern.exec(fileName).index));
        } else {
          return fileName;
        }
      }
      // lastIndexOf 实现去除文件后缀名
      function splitFileName(fileName){
        if (fileName.lastIndexOf('.') > -1) {
          var res = fileName.substring(0, fileName.lastIndexOf('.'));
          return res;
        } else {
          return fileName;
        }
      }
